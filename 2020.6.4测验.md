## 使用直线划分空间
答：1.L(n) = n*(n+1) / 2 + 1
2.
```

#include <stdio.h>

//计算直线分割平面
int calc_line_spaces(int n){
    if (n < 0) {
        return 0;
    }
    
    int result = 0;
    result = n * (n + 1) / 2 + 1;
    
    return result;
}

int main(int argc, const char * argv[]) {
    int n = 0;
    printf("请问您想使用几条直线分割平面：\n");
    scanf("%d", &n);
    
    if (calc_line_spaces(n) == 0) {
        printf("请输入大于0的正整数！");
    } else {
        printf("%d条直线所能分割的平面数量为%d\n", n, calc_line_spaces(n));
    }
    
    return 0;
}
```

## 使用折线分割平面
1.答：Z(n) = 2n<sup>2</sup> - n + 1
2.答:
```
//
//  main.c
//  2020_06_04_Test
//
//  Created by zwy on 2020/6/4.
//  Copyright © 2020 edz. All rights reserved.
//

#include <stdio.h>
#include <math.h>

//计算折线分割平面
int calc_zig_spaces(int n) {
    if (n < 0) {
        return 0;
    }
    
    int result = 0;
    result = 2 * pow(n, 2) - n + 1;
    
    return result;
}

int main(int argc, const char * argv[]) {
    int n = 0;
    printf("请问您想使用几条折线分割平面：\n");
    scanf("%d", &n);
    
    if (calc_zig_spaces(n) == 0) {
        printf("请输入大于0的正整数！");
    } else {
        printf("%d条折线所能分割的平面数量为%d\n", n, calc_zig_spaces(n));
    }
    
    return 0;
}

```

## 打印三角形
答：
```
#include <stdio.h>
#define SIZE 20


void draw(int n) {
    int temp = 0;
    int lastResult[SIZE][SIZE] = {0};// 定义存放结果的值
    int count = 0;//定义计数器
    int countArr = 0;
    
    //最外层循环搞定几行
    for (int i = 0; i < n; i++) {
        //首先打印n - 1个空格
        for (int m = i; m < n - 1; m++) {
            printf(" ");
        }
        
        //打印数字
        //处理前两行数据 对其直接打印
        if (i == 0) {
            for (int k = 0; k <= i; k++) {
                printf("%2d ", i + 1);
            }
        } else if (i == 1) {
            for (int k = 0; k <= i; k++) {
                printf("%2d ", i + 1);
                lastResult[i][countArr] = i + 1;
                countArr++;
            }
        }else {
            countArr = 0;
            temp = 0;
            for (int k = 0; k <= i; k++) {
                //第一个和最后一个数字输出为第几行的行数
                if (k == 0 || k == i) {
                    printf("%2d ", i + 1);
                    lastResult[i][countArr] = i + 1;
                    countArr++;
                } else {
                    //中间全部输出上一个数组里的两数之和
                    temp = 0;
                    count = 0;
                    temp = lastResult[i - 1][k] + lastResult[i - 1][k - 1];
                    printf("%2d ", temp);
                    lastResult[i][countArr] = temp;
                    countArr++;
                }
            }
        }
        
        printf("\n");
    }
}

int main() {
    //打印三角形
    int n = 0;
    printf("请确认打印几行的三角形:\n");
    scanf("%d", &n);
    
    draw(n);
}
```

## 实现atof函数
答：
```
#include <stdio.h>
#include <string.h>
#include <math.h>
#define SIZE 20

//扫描字符串，遇上数字或者.开始转换为double型,遇到非数字或'\0'结束转换
double my_atof (char* a) {
    char b[SIZE];
    double result = 0;
    int count = 1;
    int flag = 0;
    strcpy(b, a);
    for (int i = 0; i < strlen(b); i++) {
        if ((b[i] - '0' >= 0 && b[i] - '0' <= 9) || b[i] == '.') {
            if (b[i] == '.') {
                flag = 1;
                continue;
            }
            while (flag) {
                result = result + (b[i] - '0') * pow(10, -count);
                i++;
                count++;
                if (i >= strlen(b)) {
                    return result;
                }
            }
            result = result * 10 + (b[i] - '0');
        } else{
            return result;
        }
    }
    
    return result;
}

int main() {
    char a[SIZE];
    printf("请输入想要转换的字符串:\n");
    scanf("%s", a);
    
    printf("%.4f\n", my_atof(a));
}
```

```
void enqueue(Queue* pQueue, int data) {
```
